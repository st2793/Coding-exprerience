# 최초풀이1(직접하드코딩)
```python
import sys
sys.stdin=open('a.txt','rt')
n,x=map(int,sys.stdin.readline().rstrip().split())
nums=list(map(int,sys.stdin.readline().rstrip().split()))
lt=0
rt=n
LP=None
while lt<=rt<n:
    midL = (lt + rt)//2
    print(lt,rt,midL)
    if nums[midL]<x:
        lt=midL+1
    elif nums[midL]>x:
        rt=midL-1
    elif nums[midL]==x:
        rt-=1
        LP=midL
lt=0
rt=n
RP=None
while lt<=rt<n:
    midR = (lt + rt)//2
    if nums[midR]<x:
        lt=midR+1
    elif nums[midR]>x:
        rt=midR-1
    elif nums[midR]==x:
        lt+=1
        RP=midR
if RP==None:
    print(-1)
else:
    print(RP-LP+1)
```

# 최초풀이2(bisect활용)

```python
import sys
import bisect
sys.stdin=open('a.txt','rt')
n,x=map(int,sys.stdin.readline().rstrip().split())
nums=list(map(int,sys.stdin.readline().rstrip().split()))

l=bisect.bisect_left(nums,x)
r=bisect.bisect_right(nums,x)
if r-l==0:
    print(-1)
else:
    print(r-l)
```
